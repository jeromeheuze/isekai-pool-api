Quick API

# All nodes health
curl https://api.isekai-pool.com/api/v1/health

# Yenten block count
curl -X POST https://api.isekai-pool.com/api/v1/yenten/rpc \
  -H "Content-Type: application/json" \
  -d '{"method":"getblockcount","params":[]}'

# Koto blockchain info
